<head>

    <title>Market Cap Calculator</title>

</head>

<body>

    <h1>Market Cap Calculator</h1>

    <form>

        <label for="price">Price per coin ($):</label>

        <input type="number" id="price" step="0.01"><br><br>

        <label for="supply">Circulating supply:</label>

        <input type="number" id="supply"><br><br>

        <button type="button" onclick="calculate()">Calculate</button>

    </form>

    <p id="result"></p>

    <script>

        function calculate() {

            const price = parseFloat(document.getElementById("price").value);

            const supply = parseFloat(document.getElementById("supply").value);

            if (!isNaN(price) && !isNaN(supply)) {

                const marketCap = price * supply;

                document.getElementById("result").textContent = `Market Cap: $${marketCap.toLocaleString()}`;

            } else {

                document.getElementById("result").textContent = "Please enter valid numbers.";

            }

        }

    </script>

</body>
